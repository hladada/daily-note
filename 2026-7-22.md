# 1.初始化仓库
### 1.1 在项目文件夹执行
- git init
- git remote add origin git@xxx.git

# 2.提交内容到本地仓库
### 追踪所有的变更的文件
- git add .
### 提交说明
- git commit -m "初始提交"

# 3.推送本地仓库至远程仓库
### 推送main分支，建立关联
- git push -u origin main

# 4.创建新分支：feat-001
- git branch feat-001

# 5.切换到分支：feat-001
- git checkout feat-001

# 6. 在 feat-001 提交内容并推送到本地仓库
### 修改文件后
- git add .
- git commit -m "feat: 001新增功能"

# 7.将feat-001 推送到远程仓库
- git push -u origin feat-001

# 8.发起 PR，合并 feat-001 内容到 origin/main
- git checkout main
- git pull origin main

# 9. 在本地删除 feat-001，并同步仓库
#### 先切回main分支，不能在待删除分支内执行
- git checkout main
#### 删除本地分支
- git branch -d feat-001
#### 删除远程分支
- git push origin --delete feat-001
